{{- /* partials/author.html — Resolve a page's author from a content file. Reads .Params.author (a slug string) from the passed page context and looks up the corresponding author page at /authors/. Example frontmatter: author: maya-chen The author's details live in content/authors/maya-chen/_index.md: --- name: "Maya Chen" email: "maya@example.com" # optional — used for RSS website: "https://..." # optional — used for byline link --- Note: `url` is a reserved Hugo frontmatter key, so author profile pages use `website` for the author's external URL. Returns a dict { name, url, email }. Keys are present only when set. Returns an empty dict when no author is set or the slug is not found — callers gate on `with $author.name` so posts without an author are unaffected. */ -}} {{- $result := dict -}} {{- with .Params.author -}} {{- $authorPage := site.GetPage (printf "/authors/%s" .) -}} {{- with $authorPage -}} {{- with .Params.name -}}{{- $result = merge $result (dict "name" .) -}}{{- end -}} {{- with .Params.website -}}{{- $result = merge $result (dict "url" .) -}}{{- end -}} {{- with .Params.email -}}{{- $result = merge $result (dict "email" .) -}}{{- end -}} {{- end -}} {{- end -}} {{- return $result -}}
